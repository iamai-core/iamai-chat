cmake_minimum_required(VERSION 3.15)
project(iamai-chat C CXX)

# Keep existing output directory
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

# Set C++ standard for Crow
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Windows specific settings for Crow/ASIO
if(WIN32)
    add_definitions(-D_WIN32_WINNT=0x0601)
    add_definitions(-D_CRT_SECURE_NO_WARNINGS)
    add_definitions(-DASIO_STANDALONE)
endif()

# Your existing llama.cpp subdirectory
add_subdirectory(iamai-core/llama.cpp)

# Add Crow include directory
include_directories(${CMAKE_SOURCE_DIR}/include)

# Your existing executable with source files
add_executable(iamai-chat iamai-chat.cpp iamai-core/core/interface.cpp)

# Link existing llama library
target_link_libraries(iamai-chat llama)

# Add your existing include directories
target_include_directories(${PROJECT_NAME} PUBLIC iamai-core)

# Find and link threads (required for ASIO/Crow)
find_package(Threads REQUIRED)
target_link_libraries(iamai-chat Threads::Threads)
